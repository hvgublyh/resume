2.在添加option之后
echarts3.setOption(option);

//设置默认选中高亮部分

var index=4;

echarts3.dispatchAction({type:'highlight',seriesIndex:0,

dataIndex:index});

echarts3.on('mouseover',function(e){

if(e.dataIndex!=index){

echarts3.dispatchAction({type:'downplay',seriesIndex:0,

dataIndex:index});

}else{

echarts3.dispatchAction({type:'downplay',seriesIndex:0,

dataIndex:0});

}

});
